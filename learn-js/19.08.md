 const dispatch = () => {
  return new Promise(function (resolve, reject, url) {
    fetch(url).then(function (res) {
      resolve(res);
    });
  });
};

(function () {
  let times = 5;
  return new Promise(function (resolve, reject) {
    function attempt() {
      Promise.resolve()
        .then(dispatch)
        .then((res) => {
          if (times === 0) {
            reject();
          } else if (res.status === 200 && res.ok) {
            console.log("resolved");
            resolve(res);
          } else {
            times--;
            console.log(times);
            attempt();
          }
        });
    }
    attempt();
  });
})();
